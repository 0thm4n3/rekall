 19  """ 
 20  This module defines all the components that Rekall knows about. 
 21   
 22  A component is a collection of properties that relate to an entity, which is 
 23  an encapsulated notion of identity. In Rekall, components are basically just 
 24  named tuples which we store in a big hashtable, indexed by the entity they 
 25  relate to. 
 26   
 27  See: 
 28    http://en.wikipedia.org/wiki/Entity_component_system 
 29  """ 
 30  __author__ = "Adam Sindelar <adamsh@google.com>" 
 31
