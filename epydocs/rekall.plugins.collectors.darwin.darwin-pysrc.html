 77      # /xnu-1699.26.8/bsd/kern/kern_subr.c: 327 
 78      # hashinit(int elements, int type, u_long *hashmask) { 
 79      #    ... 
 80      # *hashmask = hashsize - 1;

 82      # Hence the value in _pgrphash is one less than the size of the hash 
 83      # table.

 86          "_pgrphashtbl", 
 87          target="Pointer",

 89              target="Array",

 91                  target="pgrphashhead", 
 92                  count=profile.get_constant_object( 
 93                      "_pgrphash", "unsigned long") + 1 
 94              ) 
 95          ) 
 96      ) 
 97
